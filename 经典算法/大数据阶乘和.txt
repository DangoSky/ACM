#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	long long nn;
	
	while(cin>>nn)
	{	
    	long long tem=0;
	    long long tt[20000];
	    memset(tt,0,sizeof(tt));
	    
		for(long long n=1;n<=nn;n++)
		{
	        long long s[20000];
         	memset(s,0,sizeof(s));
      	    s[0]=1;
      	    int t=-1,flag=0;
		    for(int i=2;i<=n;i++)
		      {
		    	int k=0;
			    for(int j=0;j<=flag;j++)
			     {
			 	    long long a=s[j]*i+k;
			 	    s[j]=a%10;
			 	    k=a/10;
			     }
			    while(k)
			     {
			 	   s[++flag]=k%10;
			 	   k=k/10;
			     }
		      }
		      //½×³ËÏà¼Ó 
	       if(flag>tem)  tem=flag;
		   for(int i=0;i<=flag;i++)
		    {
			    tt[i]+=s[i];
			    if(tt[i]>=10)
			     {
			    	tt[i+1]++;
				    tt[i]-=10;
				    if(i+1>tem)  tem=i+1;
			     }
	    	}
	   }
    	for(int i=tem;i>=0;i--)
		     cout<<tt[i];
	    cout<<endl;	
	}
 }
